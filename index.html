const SHEET_ID = '1fz2fBqPcp9eYtNAEhvCGwMd55JRppbpGXLByZ3J1x2A'; function toNum(val) { if (val === '' || val === null || val === undefined) return val; const str = val.toString().replace(',', '.'); const num = parseFloat(str); return isNaN(num)? val : num; } function makeJsonOutput(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } // CARI BARIS KOSONG PERTAMA DARI ATAS function getFirstEmptyRow(sheet) { const maxRows = sheet.getMaxRows(); const values = sheet.getRange(2, 1, maxRows-1, 1).getValues(); for (let i = 0; i < values.length; i++) { if (values[i][0] === '' || values[i][0] === null) return i + 2; } return sheet.getLastRow() + 1; } // AMBIL DATA NAMA PRODUK DARI MASTER KOLOM "nama produk" function getMasterProduk() { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName('MASTER'); if (!sheet) return []; const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const namaIdx = headers.indexOf('nama produk'); if (namaIdx < 0) return []; const values = sheet.getRange(2, namaIdx+1, sheet.getLastRow()-1, 1).getValues(); return values.flat().filter(String); } function doPost(e) { const ss = SpreadsheetApp.openById(SHEET_ID); const data = JSON.parse(e.postData.contents); const sheet = ss.getSheetByName(data._sheet); if (!sheet) return makeJsonOutput({status:'ERROR', msg:'Sheet not found'}); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; let insertedRows = []; let pdfUrls = []; // MODE KIRIM BANYAK PRODUK SEKALIGUS if (data.items && Array.isArray(data.items)) { data.items.forEach(function(item) { const targetRow = getFirstEmptyRow(sheet); const rowData = { 'Tanggal': data.Tanggal, 'Nomer Surat Jalan': data['Nomer Surat Jalan'], 'Nama Barang': item['Nama Barang'], 'Kirim': item['Kirim'], 'Keterangan': data.Keterangan, 'No Polisi': data['No Polisi'], 'Driver': data.Driver }; headers.forEach(function(h, idx) { if (h!== '' && rowData[h]!== undefined) { let val = rowData[h]; if (['Kirim','Box'].includes(h)) val = toNum(val); sheet.getRange(targetRow, idx + 1).setValue(val); } }); const pdfUrl = createPdfBukti(rowData, targetRow, data._sheet); insertedRows.push(targetRow); pdfUrls.push(pdfUrl); }); return makeJsonOutput({ status: 'OK', rows: insertedRows, pdfUrls: pdfUrls, waText: encodeURIComponent(`Bukti Input ${data._sheet}\nTanggal: ${data.Tanggal}\nTotal Barang: ${data.items.length}\nLink: ${pdfUrls[0]}`) }); } // MODE KIRIM 1 BARIS - untuk input Hasil Prod. const targetRow = getFirstEmptyRow(sheet); headers.forEach(function(h, idx) { if (h!== '' && data[h]!== undefined) { let val = data[h]; if (['Box','Kirim'].includes(h)) val = toNum(val); sheet.getRange(targetRow, idx + 1).setValue(val); } }); const pdfUrl = createPdfBukti(data, targetRow, data._sheet); return makeJsonOutput({ status: 'OK', row: targetRow, pdfUrl: pdfUrl, waText: encodeURIComponent(`Bukti Input ${data._sheet}\nTanggal: ${data.Tanggal}\nProduk: ${data.Produk || data['Nama Barang']}\nLink: ${pdfUrl}`) }); } function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); // UNTUK AMBIL DROPDOWN NAMA PRODUK DARI MASTER if (e.parameter.action === 'getBarang') { const list = getMasterProduk(); return makeJsonOutput({status:'OK', data: list}); } // Handler cari data untuk edit const sheet = ss.getSheetByName(e.parameter.sheet); if (!sheet) return makeJsonOutput({error:'Sheet not found'}); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const allData = sheet.getRange(2, 1, sheet.getLastRow()-1, sheet.getLastColumn()).getValues(); const tgl = e.parameter.tgl; const produk = e.parameter.produk; for (let i = 0; i < allData.length; i++) { const rowTgl = allData[i][0]; const rowProduk = allData[i][2]; // Nama Barang ada di kolom ke 3 if (Utilities.formatDate(new Date(rowTgl), Session.getScriptTimeZone(), 'yyyy-MM-dd') === tgl && rowProduk === produk) { const result = {}; headers.forEach(function(h, idx) { if (h) result[h] = allData[i][idx]; }); return makeJsonOutput(result); } } return makeJsonOutput({error:'Data tidak ditemukan'}); } // FUNGSI BIKIN PDF BUKTI function createPdfBukti(data, row, sheetName) { const ss = SpreadsheetApp.openById(SHEET_ID); const tempSheet = ss.insertSheet('TEMP_BUKTI_' + new Date().getTime()); tempSheet.getRange(1,1).setValue('BUKTI INPUT DATA'); tempSheet.getRange(2,1).setValue('Sheet:'); tempSheet.getRange(2,2).setValue(sheetName); tempSheet.getRange(3,1).setValue('Tanggal:'); tempSheet.getRange(3,2).setValue(data.Tanggal); let r = 4; for (let key in data) { if (key.startsWith('_')) continue; tempSheet.getRange(r,1).setValue(key + ':'); tempSheet.getRange(r,2).setValue(data[key]); r++; } tempSheet.getRange(r+1,1).setValue('Row: ' + row); tempSheet.getRange(r+2,1).setValue('Waktu: ' + new Date()); const url = ss.getUrl().replace('/edit', '/export?format=pdf&gid=' + tempSheet.getSheetId()); const blob = UrlFetchApp.fetch(url, { headers: { 'Authorization': 'Bearer ' + ScriptApp.getOAuthToken() } }).getBlob().setName('Bukti_' + sheetName + '_' + row + '.pdf'); const file = DriveApp.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); ss.deleteSheet(tempSheet); return file.getUrl(); }
