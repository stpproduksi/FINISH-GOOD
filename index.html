const SHEET_ID = '1fz2fBqPcp9eYtNAEhvCGwMd55JRppbpGXLByZ3J1x2A'; function toNum(val) { if (val === '' || val === null || val === undefined) return val; const str = val.toString().replace(',', '.'); const num = parseFloat(str); return isNaN(num)? val : num; } function makeOutput(text) { return ContentService.createTextOutput(text).setMimeType(ContentService.MimeType.TEXT); } function makeJsonOutput(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } // CARI BARIS KOSONG PERTAMA DARI ATAS function getFirstEmptyRow(sheet) { const maxRows = sheet.getMaxRows(); const values = sheet.getRange(2, 1, maxRows-1, 1).getValues(); for (let i = 0; i < values.length; i++) { if (values[i][0] === '' || values[i][0] === null) return i + 2; } return sheet.getLastRow() + 1; } function doPost(e) { const ss = SpreadsheetApp.openById(SHEET_ID); const data = JSON.parse(e.postData.contents); const sheet = ss.getSheetByName(data._sheet); if (!sheet) return makeOutput('Sheet not found'); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const targetRow = getFirstEmptyRow(sheet); // Simpan data ke baris kosong pertama headers.forEach(function(h, idx) { if (h!== '' && data[h]!== undefined) { let val = data[h]; // auto convert angka if (['Box','Kirim'].includes(h)) val = toNum(val); sheet.getRange(targetRow, idx + 1).setValue(val); } }); // Bikin link download PDF bukti const pdfUrl = createPdfBukti(data, targetRow); return makeJsonOutput({ status: 'OK', row: targetRow, pdfUrl: pdfUrl, waText: encodeURIComponent(`Bukti Input ${data._sheet}\nTanggal: ${data.Tanggal}\nProduk: ${data.Produk || data['Nama Barang']}\nLink: ${pdfUrl}`) }); } function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); // Handler ambil data MASTER untuk dropdown if (e.parameter.action === 'getMaster') { const sheet = ss.getSheetByName('MASTER'); if (!sheet) return makeJsonOutput({error:'MASTER not found'}); const data = sheet.getDataRange().getValues(); return makeJsonOutput({data: data}); } // Handler ambil data untuk edit const sheet = ss.getSheetByName(e.parameter.sheet); if (!sheet) return makeJsonOutput({error:'Sheet not found'}); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const allData = sheet.getRange(2, 1, sheet.getLastRow()-1, sheet.getLastColumn()).getValues(); for (let i = 0; i < allData.length; i++) { const rowTgl = allData[i][0]; const rowProduk = allData[i][3]; // kolom Produk/Nama Barang if (Utilities.formatDate(new Date(rowTgl), Session.getScriptTimeZone(), 'yyyy-MM-dd') === e.parameter.tgl && rowProduk === e.parameter.produk) { const result = {}; headers.forEach(function(h, idx) { if (h) result[h] = allData[i][idx]; }); return makeJsonOutput(result); } } return makeJsonOutput({error:'Data tidak ditemukan'}); } // FUNGSI BIKIN PDF BUKTI function createPdfBukti(data, row) { const ss = SpreadsheetApp.openById(SHEET_ID); const tempSheet = ss.insertSheet('TEMP_BUKTI_' + new Date().getTime()); tempSheet.getRange(1,1).setValue('BUKTI INPUT DATA'); tempSheet.getRange(2,1).setValue('Sheet:'); tempSheet.getRange(2,2).setValue(data._sheet); tempSheet.getRange(3,1).setValue('Tanggal:'); tempSheet.getRange(3,2).setValue(data.Tanggal); let r = 4; for (let key in data) { if (key.startsWith('_')) continue; tempSheet.getRange(r,1).setValue(key + ':'); tempSheet.getRange(r,2).setValue(data[key]); r++; } tempSheet.getRange(r+1,1).setValue('Row: ' + row); tempSheet.getRange(r+2,1).setValue('Waktu: ' + new Date()); const url = ss.getUrl().replace('/edit', '/export?format=pdf&gid=' + tempSheet.getSheetId()); const blob = UrlFetchApp.fetch(url, { headers: { 'Authorization': 'Bearer ' + ScriptApp.getOAuthToken() } }).getBlob().setName('Bukti_' + data._sheet + '_' + row + '.pdf'); const file = DriveApp.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); ss.deleteSheet(tempSheet); return file.getUrl(); }
